# CMP ResStrings 示例（含命名参数）
def get_replaced_text(res_name: str, args: dict[str, str]) -> str:
    if len(args) == 0:
        return f"ResStrings.{res_name}"
    formatter = ", ".join([f"{k} = {v}.toString()" for k, v in args.items()])
    return f"ResStrings.{res_name}.format(" + formatter + ")"

def get_import_statements(module_name: str, file_name: str) -> str:
    return f"import com.funny.translation.{module_name}.strings.ResString"
